## Doing a Strictly before and after comparisons 

## Region Sales 

```sql
WITH region_sales AS
(
SELECT region , 
SUM(CASE WHEN week_number BETWEEN 25-12 AND 25-1 THEN sales END) AS twelve_weeks_before_sales,
SUM(CASE WHEN week_number BETWEEN 26 AND 25+12 THEN sales END) AS twelve_weeks_after_sales
  FROM cleaned_weekly_sales 
WHERE calendar_year =2020
GROUP BY region
)


SELECT region, 
ROUND((100*( twelve_weeks_after_sales -twelve_weeks_before_sales  ) /NULLIF(twelve_weeks_before_sales,0)),2) :: DECIMAL AS 
 percentage_change_in_sales_for_12_week_period,
 
RANK() OVER(ORDER BY ROUND((100*( twelve_weeks_after_sales -twelve_weeks_before_sales  ) /NULLIF(twelve_weeks_before_sales,0)),2) :: DECIMAL ASC) as rank 
 
FROM region_sales

```
